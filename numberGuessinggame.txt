import java.util.Random;
import java.util.Scanner;

public class NumberGuessingGame {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();
        
        int maxNumber = 100;
        int randomNumber = random.nextInt(maxNumber) + 1;
        int attempts = 0;
        int maxAttempts = 10;
        
        System.out.println("=== Welcome to the Number Guessing Game ===");
        System.out.println("I'm thinking of a number between 1 and " + maxNumber);
        System.out.println("You have " + maxAttempts + " attempts to guess it!");
        
        while (attempts < maxAttempts) {
            System.out.print("\nEnter your guess: ");
            
            // Input validation
            if (!scanner.hasNextInt()) {
                System.out.println("Please enter a valid number!");
                scanner.next(); // Clear invalid input
                continue;
            }
            
            int guess = scanner.nextInt();
            attempts++;
            
            if (guess == randomNumber) {
                System.out.println("ðŸŽ‰ Congratulations! You guessed the number " + randomNumber + " in " + attempts + " attempts!");
                break;
            } else if (guess < randomNumber) {
                System.out.println("ðŸ“ˆ Too low! Try higher.");
            } else {
                System.out.println("ðŸ“‰ Too high! Try lower.");
            }
            
            // Show remaining attempts
            int remaining = maxAttempts - attempts;
            System.out.println("Attempts left: " + remaining);
            
            // Give hint when few attempts remain
            if (remaining <= 3 && remaining > 0) {
                if (randomNumber % 2 == 0) {
                    System.out.println("ðŸ’¡ Hint: The number is EVEN");
                } else {
                    System.out.println("ðŸ’¡ Hint: The number is ODD");
                }
            }
        }
        
        // If player runs out of attempts
        if (attempts == maxAttempts) {
            System.out.println("\nðŸ’” Game Over! The number was: " + randomNumber);
        }
        
        scanner.close();
    }
}
